//REGION CLIENT_ID
IF
GameStarted(_Region, _IsEditorMode)
THEN
NRD_ModCall("WeaponExpansion", "OnGameStarted", _Region, (STRING)_IsEditorMode);

IF
UserConnected(_ID, _, _)
AND
DB_LeaderLib_GameStarted(1)
AND
GetCurrentCharacter(_ID, _Player)
THEN
NRD_ModCall("WeaponExpansion", "SendClientID", (STRING)_Player, (STRING)_ID);

IF
UserEvent(_ID, "LLWEAPONEX_SendClientID")
AND
GetCurrentCharacter(_ID, _Player)
THEN
NRD_ModCall("WeaponExpansion", "SendClientID", (STRING)_Player, (STRING)_ID);

IF
CharacterReservedUserIDChanged(_Player, _Old, _New)
AND
_Old != _New
THEN
NRD_ModCall("WeaponExpansion", "SendClientID", (STRING)_Player, (STRING)_New);

IF
TimerFinished("Timers_LLWEAPONEX_SetActivePlayers")
THEN
IterateUsers("LLWEAPONEX_SendClientID");

IF
TimerFinished("Timers_LLWEAPONEX_OnLuaReset")
AND
DB_CurrentLevel(_Region)
AND
GlobalGetFlag("LeaderLib_IsEditorMode", _IsEditorMode)
THEN
NRD_ModCall("WeaponExpansion", "OnGameStarted", _Region, (STRING)_IsEditorMode, "1", "1");
//END_REGION

//REGION TRADE_WINDOW
IF
RequestTrade(_Player, _Npc)
THEN
LeaderLib_Timers_StartObjectTimer(_Player, 50, "Timers_LLWEAPONEX_UI_HookIntoTradeWindow", "LLWEAPONEX_UI_HookIntoTradeWindow");

IF
TradeEnds(_Player, _Npc)
THEN
LeaderLib_Timers_CancelObjectTimer(_Player, "Timers_LLWEAPONEX_UI_HookIntoTradeWindow");

IF
StoryEvent((CHARACTERGUID)_Player, "LLWEAPONEX_UI_HookIntoTradeWindow")
THEN
NRD_ModCall("WeaponExpansion", "HookIntoTradeWindow", (STRING)_Player);
//END_REGION
