PROC
LLWEAPONEX_TagHelper_CheckWeapon_Internal_TypeFound((CHARACTERGUID)_Char, (STRING)_WeaponType)
AND
DB_LLWEAPONEX_CheckWeaponType_Temp_Checking(_Char, _Slot)
AND
DB_LLWEAPONEX_TagHelper_WeaponTypeToTag(_WeaponType, _Tag)
AND
CharacterGetEquippedItem(_Char, _Slot, _Item)
THEN
DB_LLWEAPONEX_CheckWeaponType_Temp_Tagged(_Char, _Slot);
SetTag(_Item, _Tag);
SetTag(_Item, "LLWEAPONEX_TaggedWeaponType");

IF
StoryEvent((CHARACTERGUID)_Char, "LLWEAPONEX_CheckWeaponType_Done")
AND
DB_LLWEAPONEX_CheckWeaponType_Temp_EquipItem(_Char, _Item)
AND
ObjectGetFlag(_Item, "LLWEAPONEX_ItemWasLocked", 1)
THEN
NOT DB_LLWEAPONEX_CheckWeaponType_Temp_EquipItem(_Char, _Item);
ObjectClearFlag(_Item, "LLWEAPONEX_ItemWasLocked", 0);
CharacterEquipItem(_Char, _Item);
ItemLockUnEquip(_Item, 1);

IF
StoryEvent((CHARACTERGUID)_Char, "LLWEAPONEX_CheckWeaponType_Done")
AND
DB_LLWEAPONEX_CheckWeaponType_Temp_EquipItem(_Char, _Item)
THEN
NOT DB_LLWEAPONEX_CheckWeaponType_Temp_EquipItem(_Char, _Item);
CharacterEquipItem(_Char, _Item);

IF
StoryEvent((CHARACTERGUID)_Char, "LLWEAPONEX_CheckWeaponType_Done")
AND
NOT DB_LLWEAPONEX_CheckWeaponType_Temp_Tagged(_Char, _)
AND
DB_LLWEAPONEX_CheckWeaponType_Temp_Checking(_Char, _Slot)
AND
CharacterGetEquippedItem(_Char, _Slot, (ITEMGUID)_Item)
AND
String(_Item, _Str)
THEN
NOT DB_LLWEAPONEX_CheckWeaponType_Temp_Checking(_Char, _Slot);
LeaderLog_Log("DEBUG", "[LLWEAPONEX_00_Helpers:CheckWeaponType] [ERROR] Failed to find weapon type for weapon [",_Str,"] in slot [",_Slot,"]!");

IF
StoryEvent((CHARACTERGUID)_Char, "LLWEAPONEX_CheckWeaponType_Done")
AND
DB_LLWEAPONEX_CheckWeaponType_Temp_Checking(_Char, _Slot)
THEN
NOT DB_LLWEAPONEX_CheckWeaponType_Temp_Checking(_Char, _Slot);
NOT DB_LLWEAPONEX_CheckWeaponType_Temp_Tagged(_Char, _Slot);

IF
StoryEvent((CHARACTERGUID)_Char, "LLWEAPONEX_WeaponTypeFound_Axe")
THEN
