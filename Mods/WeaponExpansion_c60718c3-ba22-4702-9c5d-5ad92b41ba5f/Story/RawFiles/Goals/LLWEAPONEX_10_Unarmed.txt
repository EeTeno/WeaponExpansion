PROC
LLWEAPONEX_WeaponMastery_OnMasteryActivated((CHARACTERGUID)_Player, "LLWEAPONEX_Unarmed")
THEN

PROC
LLWEAPONEX_WeaponMastery_OnMasteryDeactivated((CHARACTERGUID)_Player, "LLWEAPONEX_Unarmed")
THEN
CharacterRemoveSkill(_Player, "Target_SinglehandedAttack");

IF
SkillDeactivated(_Player, "Target_SinglehandedAttack")
AND
LLWEAPONEX_WeaponMastery_QRY_HasActiveMastery(_Player, "LLWEAPONEX_Unarmed")
THEN
CharacterAddSkill(_Player, "Target_SinglehandedAttack", 0);
